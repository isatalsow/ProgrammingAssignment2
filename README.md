cacheMean <- function(x, ...) {
    m <- x$getmean()
    if(!is.null(m)) {
        message("getting cached data")
        return(m)
    }
    data <- x$get()
    m <- mean(data, ...)
    x$setmean(m)
    m
}

vec <- makeVector(c(1, 2, 3, 4))
cachemean(vec) # Calculates and caches the mean
cachemean(vec) # Retrieves the cached mean

}
